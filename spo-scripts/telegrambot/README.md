```
curl -s -o tbot-start.sh  https://raw.githubusercontent.com/dodopontocom/cardano-spo-tbot/develop/tbot-start.sh

./tbot-start.sh <TELEGRAM_BOT_TOKEN> <TELEGRAM_DESTINATION_CHAT_ID>
```
